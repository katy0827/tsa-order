const AVATAR_COLORS = [

let orders = {};

function getTotal(email) {
  return orders[email].reduce((s,r) => s + r.price * r.qty, 0);
}

function buildSelectHTML(selectedName) {
  let html = `<select>`;
  html += `<option value="">— 選擇品項 —</option>`;
  for (const [cat, items] of Object.entries(MENU)) {
    html += `<optgroup label="${cat}">`;
    for (const item of items) {
      const sel = item.name === selectedName ? ' selected' : '';
      html += `<option value="${item.name}|||${item.price}"${sel}>${item.name}（NT$${item.price}）</option>`;
    }
    html += `</optgroup>`;
  }
  html += `</select>`;
  return html;
}

function render() {
  let orderedCount = 0, overCount = 0, grandTotal = 0;
  const container = document.getElementById('main');
  let html = '';

  PEOPLE.forEach((person, idx) => {
    const email = person.email;
    const total = getTotal(email);
    const isOver = total > BUDGET;
    const hasOrders = orders[email].length > 0;
    if (hasOrders) orderedCount++;
    if (isOver) overCount++;
    grandTotal += total;

    const ac = AVATAR_COLORS[idx % AVATAR_COLORS.length];
    const initials = person.name.replace(/[a-z\s]/g,'').substring(0,2) || person.name.substring(0,2);
    const isOpen = expanded[email];

    let badgeClass = 'badge-empty', badgeText = '未點餐';
    if (hasOrders && isOver) { badgeClass = 'badge-over'; badgeText = `NT$${total} ▲超額`; }
    else if (hasOrders) { badgeClass = 'badge-ok'; badgeText = `NT$${total}`; }

    html += `<div class="person-card" id="card-${idx}">
      <div class="person-header" onclick="toggle('${email}')">
        <div class="avatar" style="background:${ac.bg};color:${ac.tc}">${initials}</div>
        <div class="name-block">
          <div class="name">${person.name}</div>
          <div class="email">${email}</div>
        </div>
        <span class="badge ${badgeClass}">${badgeText}</span>
        <span class="chevron${isOpen?' open':''}">▾</span>
      </div>`;

    if (isOpen) {
      html += `<div class="order-body">`;
      orders[email].forEach((row, ri) => {
        html += `<div class="order-row" data-email="${email}" data-ri="${ri}">
          <div class="sel-wrap">${buildSelectHTML(row.name)}</div>
          <div style="display:flex;align-items:center;gap:6px">
            <div class="qty-ctrl">
              <button class="qty-btn" onclick="changeQty('${email}',${ri},-1)">−</button>
              <span class="qty-val">${row.qty}</span>
              <button class="qty-btn" onclick="changeQty('${email}',${ri},1)">+</button>
            </div>
            <span class="item-price">NT$${row.price * row.qty}</span>
          </div>
          <button class="del-btn" onclick="removeItem('${email}',${ri})" title="刪除">✕</button>
        </div>`;
      });
      html += `<button class="add-btn" onclick="addItem('${email}')">＋ 新增品項</button>`;
      if (hasOrders) {
        const diff = BUDGET - total;
        html += `<div class="subtotal-row">
          ${isOver ? `<span class="exceed">超出預算 NT$${Math.abs(diff)}</span>` : `<span class="remain">剩餘額度 NT$${diff}</span>`}
          <span class="label">小計</span>
          <span class="amount" style="color:${isOver?'var(--red)':'var(--text)'}">NT$${total}</span>
        </div>`;
      }
      html += `</div>`;
    }
    html += `</div>`;
  });

  container.innerHTML = html;
  document.getElementById('s-ordered').textContent = orderedCount;
  document.getElementById('s-over').textContent = overCount;
  document.getElementById('s-total').textContent = `NT$${grandTotal.toLocaleString()}`;

  document.querySelectorAll('.order-row select').forEach(sel => {
    sel.addEventListener('change', function() {
      const row = this.closest('.order-row');
      const email = row.dataset.email;
      const ri = parseInt(row.dataset.ri);
      if (this.value) {
        const [name, price] = this.value.split('|||');
        orders[email][ri].name = name;
        orders[email][ri].price = parseInt(price);
        render();
      }
    });
  });
}

function toggle(email) { expanded[email] = !expanded[email]; render(); }
function addItem(email) { orders[email].push({name:'',price:0,qty:1}); expanded[email]=true; render(); }
function removeItem(email, idx) { orders[email].splice(idx,1); render(); }
function changeQty(email, idx, d) { orders[email][idx].qty = Math.max(1, orders[email][idx].qty + d); render(); }

function exportData() {
  let lines = ['TSA 聚餐點餐明細', '='.repeat(40), ''];
  let grand = 0;
  PEOPLE.forEach(p => {
    const total = getTotal(p.email);
    grand += total;
    lines.push(`【${p.name}】`);
    lines.push(`Email: ${p.email}`);
    if (orders[p.email].length === 0) {
      lines.push('  尚未點餐');
    } else {
      orders[p.email].forEach(r => lines.push(`  • ${r.name} ×${r.qty}  NT$${r.price * r.qty}`));
      lines.push(`  小計：NT$${total}${total > BUDGET ? `（⚠ 超額 NT$${total-BUDGET}）` : ''}`);
    }
    lines.push('');
  });
  lines.push('='.repeat(40));
  lines.push(`預估總金額：NT$${grand.toLocaleString()}`);
  const blob = new Blob([lines.join('\n')], {type:'text/plain;charset=utf-8'});
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = 'TSA聚餐點餐明細.txt';
  a.click();
}
